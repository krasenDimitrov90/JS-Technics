function Figure(units = 'cm') {
    this.units = units;
}

Figure.prototype.changeUnits = function (newUnits) {
    this.units = newUnits;
}
Figure.prototype.sayName = function () {
    console.log(this.constructor.name);
}

function Circle() {
    Figure.call(this)
}

Circle.prototype = Object.create(Figure.prototype, {
    constructor: {value: Circle, configurable: true}
});
let f = new Figure
let c = new Circle
f.sayName() - // Output - Figure
c.sayName() - // Output - Circle
console.log(c.units)  - // Output - cm
