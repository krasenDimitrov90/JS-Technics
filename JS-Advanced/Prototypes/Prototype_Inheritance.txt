function Person(name, email) {
        this.name = name;
        this.email = email;
    }

    function Teacher (name, email, subject) {
        this.subject = subject
        Person.call(this, name, email)
    }

    Teacher.prototype = Object.create(Person.prototype);

Прототипа на Teacher наследява прототипа на Person.
По този начин ползва нещата които са в Person.
Ако в последствие добавим други методи на прототипа на Teacher те няма да се закачат и за Person 

Teacher.prototype.greet = function () {
        return this.name + ' ' + 'says Hi!';
    }

let teacher = new Teacher('Donika', 'gmail', 'sub')
