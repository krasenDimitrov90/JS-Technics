В примера виждаме че result е във функцията solution.След това създаваме две променливи 
firstZeroTest и second които взимат върнатия резултат от solution който в случая е обект.
result от своя страна е стринг който след изпълнението на solution все едно се премества 
в всяка нова функция която сме приравнили към solution в случая firstZeroTest и second.
Т.е. сега във всяка една от двете можем да използваме rezult индивидуално във всяка една
от двете променливи  firstZeroTest и second.Можем да си създадем и други променливи
които да взимат резултата от solution и да го използват по същия начин


function solution() {
    let result = '';
    return {
        append(str) {
            result += str;
        },
        removeStart(num) {
            result = result.substring(num)
        },
        removeEnd(num) {
            result = result.substring(0, result.length - num)
        },
        print() {
            console.log(result);
        }
    }
}

let firstZeroTest = solution();
let second = solution();
second.append('Krasen')
second.print()

firstZeroTest.append('hello');
firstZeroTest.append('again');
firstZeroTest.removeStart(3);
firstZeroTest.removeEnd(4);
firstZeroTest.print();
