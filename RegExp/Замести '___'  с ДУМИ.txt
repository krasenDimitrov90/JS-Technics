function replaceEmptySpaces(input) {
    let text = input[0];
    let words = input[1];
    let regExp = /_+/g;
    
    while (text.includes('_')) {
        let blindWord = regExp.exec(text); - това открива първото съвпадение и го прави на масив
        let blindWordL = blindWord[0].length; - понеже е масив Взимаме Нулевия индекс
        for (let word of words) {
            let wordL = word.length;
            if (wordL === blindWordL) {
                text = text.replace(blindWord , word)
            }
        }
    }
    console.log(text);
    
}
